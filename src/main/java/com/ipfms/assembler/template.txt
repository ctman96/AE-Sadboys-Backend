package com.ipfms.assembler;

import com.ipfms.domain.model.TODO;
import org.springframework.hateoas.Link;
import org.springframework.hateoas.Resource;
import org.springframework.hateoas.ResourceAssembler;
import org.springframework.hateoas.Resources;
import org.springframework.stereotype.Component;

import java.util.List;

@Component
public class TODOResourceAssembler implements ResourceAssembler<TODO, Resource<TODO>> {
    @Override
    public Resource<TODO> toResource(TODO toDo) {
        Resource<TODO> resource = new Resource<>(toDo);
        resource.add(new Link("http://todos/" + toDo.getId()).withSelfRel());
        return resource;
    }
    public List<Resource<TODO>> toResources(List<TODO> toDos){
        List<Resource<TODO>> resources = new ArrayList<>();
        for(TODO c : toDos){
            resources.add(toResource(c));
        }
        return resources;
    }
}
